profileUser = SELECT * FROM users WHERE username = :username LIMIT 1 profilePins = SELECT p.*, u.name AS owner_name, u.username AS owner_username, u.avatar AS owner_avatar FROM pins p JOIN users u ON u.id = p.user_id WHERE u.username = :username ORDER BY p.created_at DESC profileBoards = SELECT b.*, (SELECT COUNT(*) FROM pins WHERE board_id = b.id) AS pin_count, (SELECT image FROM pins WHERE board_id = b.id ORDER BY created_at DESC LIMIT 1) AS cover_image FROM boards b WHERE b.user_id = (SELECT id FROM users WHERE username = :username) ORDER BY b.created_at DESC followerCount = SELECT COUNT(*) AS n FROM follows WHERE followed_id = (SELECT id FROM users WHERE username = :username) followingCount = SELECT COUNT(*) AS n FROM follows WHERE follower_id = (SELECT id FROM users WHERE username = :username) amFollowing = SELECT id FROM follows WHERE follower_id = :session.id AND followed_id = (SELECT id FROM users WHERE username = :username) LIMIT 1
